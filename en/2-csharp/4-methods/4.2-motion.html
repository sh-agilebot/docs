4.2.3 Converting Cartesian Position to Joint Values ​

Method NameMotion.ConvertCartToJoint( MotionPose pose , int ufIndex = 0, int tfIndex = 0 )
DescriptionConverts the pose data from Cartesian coordinates to joint values.
Request Parameterspose : MotionPose Robot pose
ufIndex : int User coordinate system ID
tfIndex : int Tool coordinate system ID
Return ValueMotionPose: Robot pose
StatusCode: Result of function execution
Compatible robot software versionCollaborative (Copper): v7.5.0.0+
Industrial (Bronze): v7.5.0.0+

Example Code

cs
using Agilebot.IR;
using Agilebot.IR.Types;
using Agilebot.IR.Motion;

public class Test
{
    public static async Task Main()
    {
        string controllerIP = "10.27.1.254";

        // Initialize the Agilebot robot
        Arm controller = new Arm(controllerIP);
        // Connect to the Agilebot robot
        StatusCode code = await controller.Connect();
        Console.WriteLine(code != StatusCode.OK ? code.GetDescription() : "Successfully connected.");

        MotionPose motionPose = new MotionPose();
        motionPose.Pt = PoseType.Cart;
        motionPose.CartData.Position = new Position
        {
            X = 300,
            Y = 300,
            Z = 300,
            A = 0,
            B = 0,
            C = 0
        };
        motionPose.CartData.Posture = new Posture
        {
            WristFlip = 1,
            ArmUpDown = 1,
            ArmBackFront = 1,
            ArmLeftRight = 1,
            TurnCircle = new List<int>(9) { 0, 0, 0, 0, 0, 0, 0, 0, 0 }
        };

        // Convert
        (MotionPose convertPose, code) = controller.Motion.ConvertCartToJoint(motionPose);
        if (code == StatusCode.OK)
        {
            Console.WriteLine(convertPose);
        }

        // Close the connection
        code = controller.Disconnect();
        Console.WriteLine(code != StatusCode.OK ? code.GetDescription() : "Successfully disconnected.");
    }
}

cs
using Agilebot.IR;
using Agilebot.IR.Types;

public class Test
{
    public static async Task Main()
    {

            Console.WriteLine(res[0][0]);
        }

        // Close the connection
        code = controller.Disconnect();
        Console.WriteLine(code != StatusCode.OK ? code.GetDescription() : "Successfully disconnected.");
    }
}
